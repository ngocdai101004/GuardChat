torch>=2.1
numpy>=1.23

# CLIP encoder vendored under vendors/SafeGuider/).

# official GuardChat release at multimedia-synergy-lab/GuardChat.
transformers>=4.43
accelerate>=0.26
huggingface_hub>=0.20
datasets>=2.18
# `safetensors` is a transitive dep but pinning prevents an annoying
# "non-contiguous tensor" save error on some BERT backbones (Issue
# #15123).
safetensors>=0.4.2

# ---------------- Gemini API (optional) ---------------
# Required ONLY for src/Gemini/. Skip if you are not running the Gemini
# rewriting baseline.
google-genai>=0.3

# ---------------- Quantisation (optional) -------------
# Required ONLY when running with `--dtype int8` or `--dtype nf4`
# (LlamaGuard, Qwen, Llama). bitsandbytes ships CUDA wheels on Linux
# and Windows; macOS users must skip 4-bit / 8-bit and use bfloat16 or
# float16 instead.
bitsandbytes>=0.43; platform_system != "Darwin"
